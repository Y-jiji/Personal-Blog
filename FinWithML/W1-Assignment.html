import matplotlib.pyplot as plt
import torch
from torch import Tensor
from torch import float
from matplotlib.animation import FuncAnimation




torch.device('cuda' if torch.cuda.is_available() else 'cpu')


fig, ax = plt.subplots()


pos_points = [
    [1, 1], [0.9, 5], [1.3, 3.9],
    [1, 4.3], [1.1, 5.1], [1.2, 5.1]
]


neg_points = [
    [1.9, 1.2], [2.05, 1], [2.1, 1.25],
    [2.2, 1.2], [2.15, 2.9], [2, 3], [1.95, 4]
]


trainset = [pos_points, neg_points]
testset = [pos_points, neg_points]


trainset[0] = torch.tensor(trainset[0]).to(float)
trainset[1] = torch.tensor(trainset[1]).to(float)
testset[0] = torch.tensor(testset[0]).to(float)
testset[1] = torch.tensor(testset[1]).to(float)




def update(step):
    w = torch.rand(2, 1)
    w.requires_grad_(True)
    w0 = torch.rand(1)
    w0.requires_grad_(True)
    lr = 1e-1
    reg_lamda = 0.01 * step


    def model(xlist):
        return (xlist @ w + w0).sigmoid()


    def criterion(predy, labely):
        _p = labely*predy.log()
        _1_p = (1-labely)*((1-predy).log())
        return -_p.sum() - _1_p.sum() + reg_lamda * (w0 * w0).sum()


    for x in range(1000):
        red_y = model(trainset[0])
        blue_y = model(trainset[1])
        criterion(red_y, 1.0).backward()
        criterion(blue_y, 0.0).backward()
        with torch.no_grad():
            w -= lr * w.grad
            w.grad = None
            w0 -= lr * w0.grad
            w0.grad = None
    print(w0)


    with torch.no_grad():
        x1, y1 = -w0.data/w[0][0].data, 0
        x2, y2 = 0, -w0.data/w[1][0].data
    ax.clear()
    ax.axline((x1, y1), (x2, y2), c='black', linestyle='--')
    ax.scatter(x=testset[0][:, 0], y=testset[0][:, 1],
               s=50, marker='+', c='r')
    ax.scatter(x=testset[1][:, 0], y=testset[1][:, 1],
               s=50, marker='_', c='b')
    ax.axis([0, 2.5, 0, 6])
    return ax




anima = FuncAnimation(
    fig,
    func=update,
    frames=100,
    interval=200
)


anima.save('../Images/regularization-on-logistic-regression.gif', dpi=500)
